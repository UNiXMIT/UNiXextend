# Pull and Run container
podman pull postgres:latest
podman run -d -p 5432:5432 --name POSTGRES -e POSTGRES_USER=support -e POSTGRES_PASSWORD=mysecretpassword -e POSTGRES_DB=support postgres:latest

# Attach to container
podman exec -it POSTGRES bash

# Container Details
Username: support
Password: mysecretpassword
DB: support

# Remove your container
podman stop POSTGRES
podman rm POSTGRES

# ODBC Driver Setup
sudo yum install postgresql-odbc

# odbc.ini
[postgres]
