# Pull and Run container
podman pull ibmcom/mq:latest
podman run --name MQ -e LICENSE=accept -e MQ_QMGR_NAME=QM1 -p 1414:1414 -p 9443:9443 -e MQ_ADMIN_PASSWORD=YourStrongPassword -d ibmcom/mq:latest

# Attach to container
podman exec -it MQ bash

# Remove your container
podman stop MQ
podman rm MQ
