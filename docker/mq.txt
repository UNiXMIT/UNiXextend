# Pull and Run container
sudo docker pull ibmcom/mq:latest
sudo docker run --name MQ -e LICENSE=accept -e MQ_QMGR_NAME=QM1 -p 1414:1414 --publish 9443:9443 -e MQ_ADMIN_PASSWORD=YourStrongPassword -d ibmcom/mq:latest

# Attach to Docker container
sudo docker exec -it MQ bash

# Remove your container
sudo docker stop MQ
sudo docker rm MQ
