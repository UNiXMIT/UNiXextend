# Pull and Run container
sudo docker pull mysql/mysql-server
sudo docker run --name=MYSQL -p 3306:3306 -d mysql/mysql-server

# Root password
sudo docker logs MYSQL 2>&1 | grep GENERATED

# Attach to Docker container
sudo docker exec -it MYSQL mysql -uroot -p

# Create user and allow remote access
mysql> CREATE USER 'support'@'localhost' IDENTIFIED BY 'YourStrongPassword';
mysql> GRANT ALL PRIVILEGES ON *.* TO 'support'@'localhost' WITH GRANT OPTION;
mysql> CREATE USER 'support'@'%' IDENTIFIED BY 'YourStrongPassword';
mysql> GRANT ALL PRIVILEGES ON *.* TO 'support'@'%' WITH GRANT OPTION;

# Remove container
sudo docker stop MYSQL
sudo docker rm MYSQL
