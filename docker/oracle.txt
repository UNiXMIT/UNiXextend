# Pull and Run container
# 11g
sudo docker pull oracleinanutshell/oracle-xe-11g
sudo docker run -d -p 1521:1521 --name ORACLE -e ORACLE_ALLOW_REMOTE=true oracleinanutshell/oracle-xe-11g

# 19c
sudo docker pull heartu41/oracle19c
sudo docker run -d -p 1521:1521 --name ORACLE -e ORACLE_PWD=Unidos30 heartu41/oracle19c

# Attach to Docker container
sudo docker exec -it ORACLE "bash"

# Container details - 11g
# sid: xe
# username: system
# password: oracle

# Container details - 19c
# sid: ORCLCDB
# username: system
# password: Unidos30

# Remove your container
sudo docker stop ORACLE
sudo docker rm ORACLE
