# Build and Run container
git clone https://github.com/oracle/docker-images.git
cd docker-images/OracleDatabase/SingleInstance/dockerfiles
sudo ./buildDockerImage.sh -v 18.4.0 -x
sudo docker run -d -p 1521:1521 --name ORACLE -e ORACLE_PWD=YourStrongPassword oracle/database:18.4.0-xe

# Attach to Docker container
sudo docker exec -it ORACLE "bash"

# Container Details
sid: XE
username: sys
password: YourStrongPassword

# Create User and Schema
sqlplus / sys as sysdba
ALTER SESSION SET "_ORACLE_SCRIPT"=true;
CREATE USER support IDENTIFIED BY YourPassword;
CREATE SCHEMA AUTHORIZATION support;

# Remove your container
sudo docker stop ORACLE
sudo docker rm ORACLE
