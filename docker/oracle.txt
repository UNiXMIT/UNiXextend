# Pull and Run container
sudo docker pull pvargacl/oracle-xe-18.4.0:latest
sudo docker run -d -p 1521:1521 --name ORACLE pvargacl/oracle-xe-18.4.0

# Attach to Docker container
sudo docker exec -it ORACLE "bash"

# Container details
# sid: xe
# username: system
# password: oracle

# Remove your container
sudo docker stop ORACLE
sudo docker rm ORACLE
