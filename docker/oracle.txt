# Pull and Run container
podman pull container-registry.oracle.com/database/free:latest
podman run -d -p 1521:1521 --name ORACLE -e ORACLE_PWD=YourStrongPassword -e ORACLE_CHARACTERSET=AL32UTF8 container-registry.oracle.com/database/free:latest

# Attach to container
podman exec -it ORACLE bash

# Container Details
sid: FREE
username: sys
password: YourStrongPassword

# Create User and Schema
podman exec -it ORACLE sqlplus sys/YourStrongPassword@//localhost:1521/FREE as sysdba
podman exec -it ORACLE sqlplus system/YourStrongPassword@//localhost:1521/FREE
podman exec -it ORACLE sqlplus pdbadmin/YourStrongPassword@//localhost:1521/FREEPDB1

ALTER SESSION SET "_ORACLE_SCRIPT"=true;
CREATE USER support IDENTIFIED BY YourPassword;
GRANT ALL PRIVILEGES TO support;

# Remove your container
podman stop ORACLE
podman rm ORACLE
