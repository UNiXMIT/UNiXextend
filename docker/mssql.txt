# Pull and Run container
podman pull mcr.microsoft.com/mssql/server:2022-latest
podman run -e "ACCEPT_EULA=Y" -e "SA_PASSWORD=YourStrongPassword" -p 1433:1433 --name MSSQL -d mcr.microsoft.com/mssql/server:2022-latest

# Attach to container
podman exec -it MSSQL bash

# Container Details
username: SA
password: YourStrongPassword

# Create User and Database
CREATE DATABASE support
GO
CREATE LOGIN support WITH PASSWORD='YourStrongPassword', DEFAULT_DATABASE=support
GO
ALTER SERVER ROLE sysadmin ADD MEMBER support
GO
USE support
GO
CREATE USER support FOR LOGIN support
GO
ALTER USER support WITH DEFAULT_SCHEMA=dbo
GO
ALTER ROLE db_owner ADD MEMBER support
GO
-- Clean up
USE master
GO
DROP DATABASE [NewDatabase]
GO
DROP LOGIN [NewLogin]
GO

# Remove container
podman stop MSSQL
podman rm MSSQL
