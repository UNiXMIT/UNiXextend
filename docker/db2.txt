# Pull and Run container
podman pull ibmcom/db2
podman run -itd --name DB2 --privileged=true -p 50000:50000 -e LICENSE=accept -e DB2INSTANCE=support -e DB2INST1_PASSWORD=YourStrongPassword -e DBNAME=support ibmcom/db2

# Attach to container
podman exec -it DB2 bash

# Remove your container
podman stop DB2
podman rm DB2
